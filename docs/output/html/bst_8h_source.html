187 template <class Comparable>
188 bool BST<Comparable>::remove(const Comparable& x, BinaryNode<Comparable>* & t) const {
189  if(t == NULL)
190  return false; // Item not found; do nothing
191  if(x < t->element)
192  return remove(x, t->left);
193  else if(t->element < x)
194  return remove(x, t->right);
195  else if(t->left != NULL && t->right != NULL) { // Two children
196  t->element = findMin(t->right)->element;
197  return remove(t->element, t->right);
198  }
199  else {
200  BinaryNode<Comparable>* oldNode = t;
201  t = (t->left != NULL) ? t->left : t->right;
202  delete oldNode;
203  return true;
204  }
205 }

286 template <class Comparable>
287 void BSTItrPost<Comparable>::advance () {
288  itrStack.pop();
289  visitStack.pop();
290  if ((!itrStack.empty()) && (visitStack.top() == false)) {
291  visitStack.pop();
292  visitStack.push(true);
293  slideDown(itrStack.top()->right);

297 template <class Comparable>
298 void BSTItrPost<Comparable>::slideDown(BinaryNode<Comparable>* n) {
299  while (n) {
300  itrStack.push(n);
301  if (n->left) {
302  visitStack.push(false);
303  n = n->left;
304  } else if (n->right) {
305  visitStack.push(true);
306  n = n->right;
307  } else {
308  visitStack.push(true);
309  break;
310  }
311  }

335 {
336  BinaryNode<Comparable>* actual = itrStack.top();
337  BinaryNode<Comparable>* next = actual->left;
